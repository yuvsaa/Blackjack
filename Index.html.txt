    <script>
        let deck = [];
        let playerHand = [];
        let dealerHand = [];
        let bet = 0;
        let playerMoney = 100;

        // Function to create a deck of cards
        function createDeck() {
            const suits = ['Hearts', 'Diamonds', 'Clubs', 'Spades'];
            const values = ['2', '3', '4', '5', '6', '7', '8', '9', '10', 'Jack', 'Queen', 'King', 'Ace'];
            for (let suit of suits) {
                for (let value of values) {
                    deck.push({ value, suit });
                }
            }
            // Shuffle the deck using Fisher-Yates algorithm
            shuffleDeck();
        }

        // Shuffle function
        function shuffleDeck() {
            for (let i = deck.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [deck[i], deck[j]] = [deck[j], deck[i]];
            }
        }

        // Initialize game
        function initializeGame() {
            createDeck();
            // Other game initialization code
            updateDisplay();
        }

        // Update the display
        function updateDisplay() {
            document.getElementById('player-hand').textContent = JSON.stringify(playerHand);
            document.getElementById('dealer-hand').textContent = JSON.stringify(dealerHand);
        }

        // Bet deduction logic
        document.getElementById('bet-button').addEventListener('click', () => {
            const betInput = document.getElementById('bet-input').value;
            bet = Math.max(0, Math.min(playerMoney, betInput));
            playerMoney -= bet;
            updateDisplay();
        });

        // Simplified endGame function
        function endGame() {
            // Logic for ending the game
            if (/* player wins */) {
                playerMoney += bet * 2;
            } else if (/* dealer wins */) {
                // Player loses bet
            } else {
                // Push condition—return bet
                playerMoney += bet;
            }
            // Reset for new game
            bet = 0;
            updateDisplay();
        }

        // Start the game
        initializeGame();
    </script>
